```python
async def main():
    manager = UserInterfaceManager()
    
    request = InterfaceRequest(
        user_id="user123",
        interface_type="web",
        device_type="desktop",
        platform="windows"
    )
    
    response = await manager.get_interface(request)
    
    if response.success:
        print(f"Interface generated: {response.interface_data['type']}")
    else:
        print(f"Error: {response.error_message}")

asyncio.run(main())
```
