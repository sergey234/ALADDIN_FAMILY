💻 Пример кода (Swift)

// Создание touch-friendly кнопки
let button = TouchFriendlyButton()
button.setTitle("ALADDIN Security", for: .normal)

// Создание touch-friendly карточки
let card = TouchFriendlyCard()
card.onTap = {
    print("Card tapped!")
}

// Применение touch-friendly эффектов
view.makeTouchFriendly(minimumSize: 50)
view.addTouchFeedback()

💻 Пример кода (Kotlin)

// Создание touch-friendly кнопки
val button = TouchFriendlyButton(context)
button.text = "ALADDIN Security"

// Создание touch-friendly карточки
val card = TouchFriendlyCard(context)
card.setOnTapListener {
    println("Card tapped!")
}

// Применение touch-friendly эффектов
view.makeTouchFriendly(50f)
view.addTouchFeedback()
